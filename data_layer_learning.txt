#data layer
transform_param中：
  scale是用来进行数据归一化的，1/255=0.00398625
  mirror用于进行镜像，实现数据增强（数据量翻倍），1表示打开
  crop_size是表示对原有数据进行剪裁的大小，训练阶段是随机剪裁的，在测试阶段是从中间
  进行裁剪的
data_param中：
  source数据源地址，lmdb数据库格式
  batch_size是每次批处理的个数
  backend选用数据的名称
  #数据层代码如下
  #使用LMDB方式的数据层
  layer{
    name:"mnist"
    type:"Data"
    top:"data"
    top:"label"
    include{
      phase:TRAIN
    }
    transform_param{
    scale:0.00390625
    mirror:1
    crop_size:227
    }
    data_param{
    source:"path/mnist_train_lmdb"
    batch_size:64
    backend:LMDB
    }
  }
  #使用HDF5数据源
  layer{
    name:"mnist"
    type:"HDF5Data"
    top:"data"
    top:"label"
    hdf5_data_param{
    source:"path/train.txt"
    batch_size:64
    backend:HDF5
    }
  }
  #使用图片数据源
  layer{
    name:"mnist"
    type:"ImageData"
    top:"data"
    top:"label"
    transform_param{
    mirror:0
    crop_size:227
    mean_file:"path/xx.binaryproto"
    }
    image_data_param{
    source:"path/list.txt"
    batch_size:64
    new_height:256
    new_weight:256
    }
  }
